    scss/
    ├── style.scss              # 진입점 — 로드 순서 정의
    ├── 1-settings/
    │   ├── _variables.scss     # Bootstrap 오버라이드 변수 (문서화 원본)

    │   ├── _mixin.scss         # 팀 믹스인 5종 (flex, ellipsis, position, drop-shadow, ani)
    │   └── _breakpoints.scss   # respond-to() 반응형 믹스인 (6키워드)
    ├── 3-generic/
    │   ├── _root.scss          # CSS Custom Properties 디자인 토큰
    │   └── _vendor.scss        # Bootstrap @import 격리
    ├── 4-elements/
    │   ├── _base.scss          # 62.5% REM, body font-size
    │   ├── _font.scss          # Pretendard GOV @font-face
    │   ├── _common.scss        # word-break, a, button, .sr-only
    │   └── _focus.scss         # 포커스 스타일 (KWCAG 2.1 AA)
    ├── 5-objects/
    │   └── _layout.scss        # Bootstrap 그리드 중복 금지 원칙
    ├── 6-components/           # 프로젝트별 UI 컴포넌트 추가
    └── 7-utilities/            # 프로젝트별 유틸리티 클래스 추가
    ```

    ## 반응형 브레이크포인트

    `respond-to()` 믹스인 사용:

    ```scss
    @use '../2-tools' as tools;

    .element {
      // 모바일 기본
      font-size: 1.4rem;

      // 태블릿 이상
      @include tools.respond-to(tablet) {
        font-size: 1.6rem;
      }

      // PC 이상
      @include tools.respond-to(pc) {
        font-size: 1.8rem;
      }
    }
    ```

    | 키워드 | 범위 |
    |--------|------|
    | `mobile-only` | ~767px |
    | `tablet` | 768px~ |
    | `tablet-only` | 768px~1023px |
    | `pc-sm` | 1024px~ |
    | `pc-sm-only` | 1024px~1279px |
    | `pc` | 1280px~ |
